import React, { useState } from 'react';

const fakeAPIResponse = (question) => {
  // Simula uma resposta básica para algumas perguntas chave
  const q = question.toLowerCase().trim();

  if (!q) return "Por favor, faça uma pergunta!";

  if (q.includes('olá') || q.includes('oi')) {
    return "Olá! Como posso ajudar você hoje?";
  }

  if (q.includes('nucgpt')) {
    return "NucGPT é um projeto incrível de IA desenvolvido para facilitar seu dia a dia.";
  }

  if (q.includes('tempo') || q.includes('clima')) {
    return "Desculpe, ainda não posso fornecer informações do clima, mas logo isso será possível!";
  }

  // Resposta padrão genérica
  return "Interessante! Vou pesquisar mais sobre isso e te retorno em breve.";
};

const NucGPTChat = () => {
  const [question, setQuestion] = useState('');
  const [answer, setAnswer] = useState('');

  const handleSubmit = (e) => {
    e.preventDefault();
    const response = fakeAPIResponse(question);
    setAnswer(response);
  };

  return (
    <div className="chat-container">
      <h2>Converse com NucGPT</h2>
      <form onSubmit={handleSubmit}>
        <input
          type="text"
          placeholder="Digite sua pergunta..."
          value={question}
          onChange={(e) => setQuestion(e.target.value)}
          className="chat-input"
        />
        <button type="submit" className="chat-button">Perguntar</button>
      </form>

      {answer && (
        <div className="chat-answer">
          <strong>Resposta:</strong>
          <p>{answer}</p>
        </div>
      )}
    </div>
  );
};

export default NucGPTChat;
